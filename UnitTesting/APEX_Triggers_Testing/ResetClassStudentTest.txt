@isTest(SeeAllData=true)
public class ResetClassStudentTest {
 @isTest static void TestResetFunctionality() 
    {
        Class__c newClass = new Class__c(Name = 'XII');
        insert newClass;
        List<Student__c> stList = new List<Student__c>();
        for(integer i = 0;i < 3; i++)
        {
            Student__c st = new Student__c(Name = 'Firstname'+i,LastName__c = 'LastName'+i, Class__c = newClass.Id);
            stList.add(st);
        }
        insert stList;
        Class__c updatedClass = [SELECT Name, Custom_Status__c FROM Class__c WHERE ID = :newClass.Id][0];
        updatedClass.Custom_Status__c = 'Reset';
        Update updatedClass;
        List<Student__c> st = [SELECT Name FROM student__c where Class__c = :newClass.Id];
        integer lstSize = st.size();
        System.assertEquals(0,lstSize);
    }
    
    
}
