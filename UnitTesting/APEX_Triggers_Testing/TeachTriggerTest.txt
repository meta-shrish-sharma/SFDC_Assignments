@isTest
public class TeachTriggerTest 
{
    @testSetup static void dataSetupForTeachTriggerTest() 
    {
        Contact cnt = new Contact(Lastname = 'TestConatct');
        insert cnt;
        Class__c cls = new Class__c(Name = 'TestClass', MaxSize__c = 3, MyCount__c = 0);
        insert cls;
    }
    @isTest static void testInsertHindiTeacher_getSucessRespond() 
    {
        Teach__c newTeach = new Teach__c();
        newteach.name = 'newHindiTeacher';
        newteach.Class__c = [SELECT Id FROM Class__c WHERE Name = 'TestClass'].Id;
        newteach.Teacher__c = [SELECT Id FROM Contact WHERE Lastname = 'TestConatct'].Id;
        newteach.Subject__c = 'Hindi';
        Database.SaveResult result;
        String operationResult;
        Test.startTest();
        try
        {
            result = Database.insert(newTeach); 
        }
        catch(Exception e)
        {
            operationResult = e.getMessage();
        }
        Test.stopTest();
        system.assert(operationResult.contains('Hindi teacher insertion not allowed'));
    }
}
