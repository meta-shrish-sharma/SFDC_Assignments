public class RandomContactFactory 
{
	public static List<Contact> generateRandomContacts(integer numberOfContact, String contactName)
    {
        List<Contact> lstContact = new  List<contact>();
        for(integer i = 0; i<numberOfContact;i++)
        {
            Contact cnt = new contact(firstname = contactName+''+i);
            lstContact.add(cnt);
        }
        return lstContact;
    }
    
}
