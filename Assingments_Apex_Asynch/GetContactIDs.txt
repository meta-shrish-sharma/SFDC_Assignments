public class GetContactIDs
{
    public static void populateContactIDs(Id contactId)
    {
        Account updatedAccount = [SELECT ContactIDs__c, (SELECT Id FROM Contacts) FROM Account WHERE Id IN (SELECT AccountId  FROM Contact WHERE Id = :contactId)];
        if(updatedAccount.ContactIDs__c != null)
        {
            if(!updatedAccount.ContactIDs__c.contains(contactId))// If ContactIDs not null then check that ID already Exist or not
            {
                updatedAccount.ContactIDs__c = updatedAccount.ContactIDs__c + ','+ contactId; //If exist then update with comma separator
            }
            update updatedAccount;
        }
        if(updatedAccount.ContactIDs__c == null)
        {
            String accountContID;
            for(Contact contactAccount: updatedAccount.Contacts) // Id null then create an string of IDs
            {
                if(accountContID == null)
                {
                    accountContID =  contactAccount.Id; //First ID should be inserted without appending comma behind so that final String looks like Id1,Id2,Id3
                }
                if(!accountContID.contains(contactAccount.Id))
                {
                    accountContID = accountContID + ',' + contactAccount.Id;//After first Id insertion comma will inserted before every ID as a separator
                }   
            }
            updatedAccount.ContactIDs__c = accountContID;
            update updatedAccount;
        }
    }
}
