public class GetContactIDs
{
    public static void CountContacts(Id contactId)
    {
        Account updatedAccount = [Select ContactIDs__c, (Select Id FROM Contacts) FROM Account WHERE Id IN (SELECT Accountid  FROM Contact WHERE Id = '0032w0000029NPEAA2')];
        String accountContID;
        for(Contact contactAccount: updatedAccount.Contacts)
        {
            
            if(accountContID == null)
            {
                accountContID =  contactAccount.Id; 
                
            }
            if(!accountContID.contains(contactAccount.Id))
            {
                accountContID = accountContID + ',' + contactAccount.Id;
            } 
            
        }
        if(updatedAccount.ContactIDs__c != accountContID)
        {
            updatedAccount.ContactIDs__c = accountContID;
            update updatedAccount;
        }
        
    }
}
