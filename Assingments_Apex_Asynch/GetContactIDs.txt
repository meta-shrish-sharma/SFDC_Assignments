public class GetContactIDs
{
  public static void CountContacts(id contactId)
  {
      ID accountId = [SELECT Accountid FROM Contact WHERE id = :contactId][0].Accountid;
      List<Account> reletedAccount= [SELECT id, ContactIDs__c FROM account WHERE id = :accountId LIMIT 1];
      String accountContID = reletedAccount[0].ContactIDs__c;
      List<Contact> accountContacts = [SELECT id FROM Contact WHERE AccountId = :accountId];
      for(Contact contactAccount: accountContacts)
      {
          
          if(accountContID == null)
          {
             accountContID =  contactAccount.Id; 
              continue;
          }
          if(!accountContID.contains(contactAccount.Id))
          {
              accountContID = accountContID + ',' + contactAccount.Id;
          } 

      }
      Account updatedAccount = new Account(id = accountId);
      updatedAccount.ContactIDs__c = AccountContID;
      update updatedAccount;
      System.debug( 'FetchedContactID : ' + updatedAccount.id);
      System.debug( 'ContactIDs : ' + AccountContID);
      System.debug( 'AcctID : ' + accountId);
  }
}
