trigger femaleMembersExist on Class__c (before delete) 
{
	List<AggregateResult> classList = [select class__r.name from student__c where sex__c = 'F' group by class__r.name having count(id) > 1];
   	List<string> classListString = new List<string>();
    for(AggregateResult ar: classList)
    {
        classListString.add(ar.get('Name')+'');
    }
    List<Class__c> classUpdated = trigger.new;
    for(Class__c currClass:classUpdated)
    {
        if(classListString.contains(currClass.Name))
        {
            currClass.addError('Cannot delete this class');
        }
    }
}
