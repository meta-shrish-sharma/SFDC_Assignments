trigger contactTriggerNotHindi1 on Contact (before insert,before update) 
{

   List<contact> updatedContact = trigger.new;
    for(contact tchUpdated: updatedContact)
    {
         List<contact> teachHindi = [select ID from contact where ID IN (select teacher__c from teach__c where subject__c IN ('Hindi')) AND ID = :tchUpdated.id];
        if(teachHindi.size() > 0)
        {
            tchUpdated.addError('Hindi teacher insertion not allowed');
        }
    }
}
