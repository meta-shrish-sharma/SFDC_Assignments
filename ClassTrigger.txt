//In delete triggers data can be access using Trigger.old 
trigger ClassTrigger on Class__c (before delete, after update) 
{
    if(trigger.isDelete && trigger.isBefore)
    {
        List<AggregateResult> classList = [SELECT Class__c FROM Student__c WHERE Sex__c = 'F' AND Class__c IN :Trigger.old GROUP BY Class__c HAVING COUNT(Id) > 1];
        List<Id> classIdList = new List<Id>();
        for(AggregateResult ar: classList)
        {
            classIdList.add((Id)ar.get('Class__c'));
        }
        for(Class__c currClass: Trigger.old)
        {
            if(classIdList.contains(currClass.Id))
            {
                currClass.addError('Cannot delete this class, This class has more then one female members');
            }
        }
    }
    if(trigger.isUpdate && trigger.isAfter)
    {
        List<Student__c> deleteStudent = [SELECT Id, Name FROM Student__c WHERE Class__c IN :Trigger.new AND Class__r.Custom_Status__c = 'Reset'];
        delete deleteStudent;
    }
}
