{!REQUIRESCRIPT("/soap/ajax/15.0/connection.js")}
{!REQUIRESCRIPT("/soap/ajax/15.0/apex.js")}

var records = {!GETRECORDIDS( $ObjectType.Student__c )}; 
if (records[0] == null) 
{ 
	alert("Please select at least one record.") 
} 
else 
{
	if(confirm("Are you sure to delete the records. Total Records: " + records.length))
	{
		var result = sforce.apex.execute("DeleteCheckedStudents","deleteStudents",{stId:records});
		if(result = "Success")
		{
			alert("Students deleted successfully");
		}
		else
		{
			alert("Failed to delete students");
		}
		window.location.reload(); 
	}
}
{
	window.location.reload(); 
}
